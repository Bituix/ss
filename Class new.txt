CLASS zcl_ca_statusmanager DEFINITION
  PUBLIC
  FINAL
  CREATE PUBLIC.

  "----------------------------------------------------------------------
  " Z-Status Framework — Central Status Manager
  "
  " Purpose : Single entry point for all status transitions.
  "           Enforces read-validate-write atomicity per (status_type,
  "           object_key) via enqueue lock EZCASTAT_LOCK.
  "
  " Lock note: The lock does NOT protect ZCASTAT_LOG write integrity —
  "            UUID PKs guarantee collision-free parallel inserts.
  "            It serialises the read-validate-write decision window so
  "            two concurrent callers for the same object instance cannot
  "            both read the same current status and produce two valid
  "            but conflicting log entries.
  "
  " Prerequisites:
  "   EZCASTAT_LOCK — SE11 lock object
  "                   fields: STATUS_TYPE + OBJECT_KEY
  "   ZCASTAT_NR    — SNRO number range, interval 01, buffered size 10
  "   BADI_ZCASTAT_TRANSITION — SE19, filter ZDE_CA_STAT_TYPE (CHAR 40)
  "----------------------------------------------------------------------

  PUBLIC SECTION.

    TYPES:
      tt_log     TYPE STANDARD TABLE OF zcastat_log     WITH EMPTY KEY,
      tt_actions TYPE STANDARD TABLE OF zcastat_action  WITH EMPTY KEY.

    "& Execute a status transition.
    "& iv_from_status: optional — auto-resolved from last log entry.
    "& iv_to_status  : optional — auto-resolved when exactly one active
    "&                 action matches iv_action_code. Raises
    "&                 AMBIGUOUS_ACTION when two or more qualify.
    METHODS change_status
      IMPORTING
        iv_status_type TYPE zde_ca_stat_type
        iv_object_key  TYPE zde_ca_stat_obj_key
        iv_action_code TYPE char20
        iv_to_status   TYPE zde_ca_stat_code OPTIONAL
        iv_from_status TYPE zde_ca_stat_code OPTIONAL
        iv_remark      TYPE char255          OPTIONAL
      RAISING
        zcx_ca_status_error.

    "& Return the current status code for a business object instance.
    "& Returns initial value when no log entry exists yet.
    METHODS get_current_status
      IMPORTING
        iv_status_type   TYPE zde_ca_stat_type
        iv_object_key    TYPE zde_ca_stat_obj_key
      RETURNING
        VALUE(rv_status) TYPE zde_ca_stat_code.

    "& Return full transition log ordered chronologically ascending.
    METHODS get_log
      IMPORTING
        iv_status_type TYPE zde_ca_stat_type
        iv_object_key  TYPE zde_ca_stat_obj_key
      RETURNING
        VALUE(rt_log)  TYPE tt_log.

    "& Return all active actions available from a given status.
    "& Use to build action buttons in the consuming application.
    METHODS get_available_actions
      IMPORTING
        iv_status_type    TYPE zde_ca_stat_type
        iv_from_status    TYPE zde_ca_stat_code
      RETURNING
        VALUE(rt_actions) TYPE tt_actions.

    "& Dry-run check: returns true when the transition would be allowed.
    "& Does not acquire the lock. Use for UI button enablement only.
    "& Only catches ZCX_CA_STATUS_ERROR — system exceptions propagate.
    METHODS is_transition_allowed
      IMPORTING
        iv_status_type    TYPE zde_ca_stat_type
        iv_object_key     TYPE zde_ca_stat_obj_key
        iv_action_code    TYPE char20
        iv_to_status      TYPE zde_ca_stat_code OPTIONAL
      RETURNING
        VALUE(rv_allowed) TYPE abap_bool.

  PRIVATE SECTION.

    "-- Internal transfer structure: resolved action row --
    TYPES: BEGIN OF ty_action,
             from_status          TYPE zde_ca_stat_code,
             to_status            TYPE zde_ca_stat_code,
             action_code          TYPE char20,
             requires_comment     TYPE xfeld,
             authorization_object TYPE char10,
           END OF ty_action.

    METHODS acquire_lock
      IMPORTING
        iv_status_type TYPE zde_ca_stat_type
        iv_object_key  TYPE zde_ca_stat_obj_key
      RAISING
        zcx_ca_status_error.

    METHODS release_lock
      IMPORTING
        iv_status_type TYPE zde_ca_stat_type
        iv_object_key  TYPE zde_ca_stat_obj_key.

    METHODS read_current_status
      IMPORTING
        iv_status_type   TYPE zde_ca_stat_type
        iv_object_key    TYPE zde_ca_stat_obj_key
      RETURNING
        VALUE(rv_status) TYPE zde_ca_stat_code.

    "& Reads ZCASTAT_ACTION once and returns the resolved action row.
    "& Raises NO_VALID_ACTION or AMBIGUOUS_ACTION when resolution fails.
    METHODS resolve_action
      IMPORTING
        iv_status_type   TYPE zde_ca_stat_type
        iv_from_status   TYPE zde_ca_stat_code
        iv_action_code   TYPE char20
        iv_to_status     TYPE zde_ca_stat_code OPTIONAL
      RETURNING
        VALUE(rs_action) TYPE ty_action
      RAISING
        zcx_ca_status_error.

    METHODS check_authorization
      IMPORTING
        iv_status_type TYPE zde_ca_stat_type
        is_action      TYPE ty_action
      RAISING
        zcx_ca_status_error.

    METHODS call_badi_validate
      IMPORTING
        io_badi        TYPE REF TO badi_zcastat_transition
        iv_status_type TYPE zde_ca_stat_type
        iv_object_key  TYPE zde_ca_stat_obj_key
        iv_from_status TYPE zde_ca_stat_code
        iv_to_status   TYPE zde_ca_stat_code
      RAISING
        zcx_ca_status_error.

    METHODS call_badi_after
      IMPORTING
        io_badi        TYPE REF TO badi_zcastat_transition
        iv_status_type TYPE zde_ca_stat_type
        iv_object_key  TYPE zde_ca_stat_obj_key
        iv_from_status TYPE zde_ca_stat_code
        iv_to_status   TYPE zde_ca_stat_code
        iv_log_uuid    TYPE sysuuid_16.

    METHODS write_log
      IMPORTING
        iv_log_uuid    TYPE sysuuid_16
        iv_status_type TYPE zde_ca_stat_type
        iv_object_key  TYPE zde_ca_stat_obj_key
        is_action      TYPE ty_action
        iv_remark      TYPE char255 OPTIONAL.

    METHODS get_next_log_number
      RETURNING
        VALUE(rv_number) TYPE int8.

ENDCLASS.


CLASS zcl_ca_statusmanager IMPLEMENTATION.

  METHOD change_status.
    "-- ① Serialise the decision window for this object instance -------
    acquire_lock(
      iv_status_type = iv_status_type
      iv_object_key  = iv_object_key ).

    "-- ② Determine where we are coming from ---------------------------
    DATA(lv_from) = COND zde_ca_stat_code(
      WHEN iv_from_status IS NOT INITIAL
      THEN iv_from_status
      ELSE read_current_status(
             iv_status_type = iv_status_type
             iv_object_key  = iv_object_key ) ).

    "-- ③ Resolve the action row (single ZCASTAT_ACTION read) ----------
    DATA(ls_action) = resolve_action(
      iv_status_type = iv_status_type
      iv_from_status = lv_from
      iv_action_code = iv_action_code
      iv_to_status   = iv_to_status ).

    "-- ④ Authority check ----------------------------------------------
    check_authorization(
      iv_status_type = iv_status_type
      is_action      = ls_action ).

    "-- ⑤ BAdI pre-condition (blocking) --------------------------------
    DATA lo_badi TYPE REF TO badi_zcastat_transition.
    GET BADI lo_badi FILTERS status_type = iv_status_type.

    call_badi_validate(
      io_badi        = lo_badi
      iv_status_type = iv_status_type
      iv_object_key  = iv_object_key
      iv_from_status = lv_from
      iv_to_status   = ls_action-to_status ).

    "-- ⑥ Append log entry ---------------------------------------------
    DATA(lv_uuid) = cl_system_uuid=>create_uuid_x16_static( ).

    write_log(
      iv_log_uuid    = lv_uuid
      iv_status_type = iv_status_type
      iv_object_key  = iv_object_key
      is_action      = ls_action
      iv_remark      = iv_remark ).

    "-- ⑦ BAdI side effect (non-blocking) ------------------------------
    call_badi_after(
      io_badi        = lo_badi
      iv_status_type = iv_status_type
      iv_object_key  = iv_object_key
      iv_from_status = lv_from
      iv_to_status   = ls_action-to_status
      iv_log_uuid    = lv_uuid ).

    "-- ⑧ Release lock -------------------------------------------------
    release_lock(
      iv_status_type = iv_status_type
      iv_object_key  = iv_object_key ).

  ENDMETHOD.


  METHOD get_current_status.
    rv_status = read_current_status(
      iv_status_type = iv_status_type
      iv_object_key  = iv_object_key ).
  ENDMETHOD.


  METHOD get_log.
    SELECT *
      FROM zcastat_log
      WHERE status_type = @iv_status_type
        AND object_key  = @iv_object_key
      ORDER BY log_number ASCENDING
               changed_at  ASCENDING
      INTO TABLE @rt_log.
  ENDMETHOD.


  METHOD get_available_actions.
    SELECT *
      FROM zcastat_action
      WHERE status_type = @iv_status_type
        AND from_status = @iv_from_status
        AND is_active   = @abap_true
      INTO TABLE @rt_actions.
  ENDMETHOD.


  METHOD is_transition_allowed.
    rv_allowed = abap_false.
    TRY.
        DATA(lv_current) = read_current_status(
          iv_status_type = iv_status_type
          iv_object_key  = iv_object_key ).

        resolve_action(
          iv_status_type = iv_status_type
          iv_from_status = lv_current
          iv_action_code = iv_action_code
          iv_to_status   = iv_to_status ).

        rv_allowed = abap_true.

      CATCH zcx_ca_status_error.
        rv_allowed = abap_false.
        "-- System exceptions intentionally not caught here.
        "-- A CX_SY_* or similar propagates to the caller as a real error.
    ENDTRY.
  ENDMETHOD.


  METHOD acquire_lock.
    CALL FUNCTION 'ENQUEUE_EZCASTAT_LOCK'
      EXPORTING
        mode_zcastat_log = 'E'
        mandt            = sy-mandt
        status_type      = iv_status_type
        object_key       = iv_object_key
        _scope           = '2'     "< auto-release on COMMIT / ROLLBACK
        _wait            = abap_false  "< fail-fast, never block caller
      EXCEPTIONS
        foreign_lock     = 1
        system_failure   = 2
        OTHERS           = 3.

    IF sy-subrc <> 0.
      RAISE EXCEPTION TYPE zcx_ca_status_error
        EXPORTING
          textid       = zcx_ca_status_error=>concurrent_lock
          status_type  = iv_status_type
          object_key   = iv_object_key
          locked_by    = sy-msgv1.
    ENDIF.
  ENDMETHOD.


  METHOD release_lock.
    CALL FUNCTION 'DEQUEUE_EZCASTAT_LOCK'
      EXPORTING
        mode_zcastat_log = 'E'
        mandt            = sy-mandt
        status_type      = iv_status_type
        object_key       = iv_object_key.
  ENDMETHOD.


  METHOD read_current_status.
    "-- Primary sort  : log_number DESC (resolves same-tick ties)
    "-- Secondary sort: changed_at DESC (fallback when log_number = 0)
    SELECT SINGLE to_status
      FROM zcastat_log
      WHERE status_type = @iv_status_type
        AND object_key  = @iv_object_key
      ORDER BY log_number DESCENDING
               changed_at  DESCENDING
      INTO @rv_status.
  ENDMETHOD.


  METHOD resolve_action.
    "-- Single read of ZCASTAT_ACTION for this from+action combination.
    "-- Result is passed through the call chain so validate and write
    "-- steps reuse it without additional DB roundtrips.
    SELECT from_status,
           to_status,
           action_code,
           requires_comment,
           authorization_object
      FROM zcastat_action
      WHERE status_type = @iv_status_type
        AND from_status = @iv_from_status
        AND action_code = @iv_action_code
        AND is_active   = @abap_true
      INTO TABLE @DATA(lt_actions).

    CASE lines( lt_actions ).
      WHEN 0.
        RAISE EXCEPTION TYPE zcx_ca_status_error
          EXPORTING
            textid       = zcx_ca_status_error=>no_valid_action
            status_type  = iv_status_type
            action_code  = iv_action_code.

      WHEN 1.
        rs_action = CORRESPONDING #( lt_actions[ 1 ] ).

      WHEN OTHERS.
        "-- Multiple active targets for this action.
        "-- Caller must supply iv_to_status to disambiguate.
        IF iv_to_status IS INITIAL.
          RAISE EXCEPTION TYPE zcx_ca_status_error
            EXPORTING
              textid       = zcx_ca_status_error=>ambiguous_action
              status_type  = iv_status_type
              action_code  = iv_action_code.
        ENDIF.

        "-- Explicit target supplied — find and validate it.
        READ TABLE lt_actions
          INTO rs_action
          WITH KEY to_status = iv_to_status.

        IF sy-subrc <> 0.
          RAISE EXCEPTION TYPE zcx_ca_status_error
            EXPORTING
              textid       = zcx_ca_status_error=>invalid_transition
              status_type  = iv_status_type
              from_status  = iv_from_status
              to_status    = iv_to_status.
        ENDIF.
    ENDCASE.
  ENDMETHOD.


  METHOD check_authorization.
    "-- Uncomment after creating auth object ZCASTAT_A in SU21.
    "-- Fields: STATTYPE CHAR 40 (status_type), ACTVT 01 (execute).
    "
    "   CHECK is_action-authorization_object IS NOT INITIAL.
    "   AUTHORITY-CHECK OBJECT is_action-authorization_object
    "     ID 'STATTYPE' FIELD iv_status_type
    "     ID 'ACTVT'    FIELD '01'.
    "   IF sy-subrc <> 0.
    "     RAISE EXCEPTION TYPE zcx_ca_status_error
    "       EXPORTING
    "         textid       = zcx_ca_status_error=>not_authorized
    "         status_type  = iv_status_type.
    "   ENDIF.
  ENDMETHOD.


  METHOD call_badi_validate.
    CALL BADI io_badi->validate_transition
      EXPORTING
        iv_status_type = iv_status_type
        iv_object_key  = iv_object_key
        iv_from_status = iv_from_status
        iv_to_status   = iv_to_status.
    "-- Any ZCX_CA_STATUS_ERROR raised by an implementation propagates
    "-- to change_status, which propagates to the caller.
    "-- Lock auto-releases on the subsequent ROLLBACK WORK.
  ENDMETHOD.


  METHOD call_badi_after.
    "-- Non-blocking: any exception is caught and logged.
    "-- The transition is already committed at this point.
    TRY.
        CALL BADI io_badi->after_transition
          EXPORTING
            iv_status_type = iv_status_type
            iv_object_key  = iv_object_key
            iv_from_status = iv_from_status
            iv_to_status   = iv_to_status
            iv_log_uuid    = iv_log_uuid.
      CATCH cx_root INTO DATA(lx).
        cl_abap_message_helper=>handle_exception( exception = lx ).
    ENDTRY.
  ENDMETHOD.


  METHOD write_log.
    INSERT zcastat_log FROM VALUE #(
      mandt        = sy-mandt
      log_uuid     = iv_log_uuid
      log_number   = get_next_log_number( )
      status_type  = iv_status_type
      object_key   = iv_object_key
      from_status  = is_action-from_status
      to_status    = is_action-to_status
      action_code  = is_action-action_code
      remark       = iv_remark
      changed_by   = sy-uname
      changed_at   = cl_abap_context_info=>get_system_date_time( )
      changed_date = sy-datum
      changed_time = sy-uzeit
    ).
  ENDMETHOD.


  METHOD get_next_log_number.
    DATA lv_number TYPE num10.
    CALL FUNCTION 'NUMBER_GET_NEXT'
      EXPORTING
        nr_range_nr             = '01'
        object                  = 'ZCASTAT_NR'
      IMPORTING
        number                  = lv_number
      EXCEPTIONS
        number_range_not_intern = 1
        object_not_found        = 2
        quantity_is_0           = 3
        quantity_is_not_1       = 4
        interval_not_found      = 5
        period_not_found        = 6
        blocks_not_found        = 7
        no_object_reference     = 8
        OTHERS                  = 9.
    rv_number = COND #( WHEN sy-subrc = 0 THEN lv_number ELSE 0 ).
    "-- sy-subrc <> 0 means number range unavailable.
    "-- Log entry still written with log_number = 0.
    "-- changed_at DESC remains the fallback sort key.
  ENDMETHOD.

ENDCLASS.
